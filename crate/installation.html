CrateInstallation

Installation

Adding Dependency

In order to use Crate you will need to download it using a BuildTool like Maven or Gradle.

Gradle Kotlin

Add Repository

build.gradle.kts
repositories {
    mavenCentral()
}

Add Dependency

build.gradle.kts
dependencies {
    implementation("io.github.milkdrinkers:crate:VERSION")
}

Shade Dependency

build.gradle.kts
plugins {
    id("com.gradleup.shadow") version "8.3.6"
}
 
tasks {
    shadowJar {
        relocate("io.github.milkdrinkers.Crate", "yourpackage.yourname.crate")
    }
}

Gradle Groovy

Add Repository

build.gradle
repositories {
    mavenCentral()
}

Add Dependency

build.gradle
dependencies {
    implementation 'io.github.milkdrinkers:crate:VERSION'
}

Shade Dependency

build.gradle
plugins {
    id 'com.gradleup.shadow' version '8.3.6'
}
 
tasks {
    shadowJar {
        relocate 'io.github.milkdrinkers.Crate', 'yourpackage.yourname.crate'
    }
}

Maven

Add Dependency

pom.xml
<dependency>
    <groupId>io.github.milkdrinkers</groupId>
    <artifactId>crate</artifactId>
    <version>VERSION</version>
</dependency>

Shade Dependency

pom.xml
<plugin>
    <groupId>org.apache.maven.plugins</groupId>
    <artifactId>maven-shade-plugin</artifactId>
    <version>3.5.0</version>
    <executions>
        <execution>
            <phase>package</phase>
            <goals>
                <goal>shade</goal>
            </goals>
        </execution>
    </executions>
    <configuration>
        <createDependencyReducedPom>false</createDependencyReducedPom>
        <relocations>
            <relocation>
                <pattern>io.github.milkdrinkers.Crate</pattern>
                <shadedPattern>yourpackage.yourname.crate</shadedPattern>
            </relocation>
        </relocations>
    </configuration>
</plugin>
