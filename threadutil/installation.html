ThreadutilInstallation

Installation

Adding Dependency

In order to use ThreadUtil you will need to download it using a BuildTool like Maven or Gradle. We also highly recommend relocating/shading the library into your own package.

You may add whichever platform adapter you need. In this example we add the bukkit adapter for Bukkit, Spigot and Paper support.

Gradle Kotlin DSL

Add Repository

build.gradle.kts
repositories {
    mavenCentral()
}

Add Dependency

build.gradle.kts
dependencies {
    implementation("io.github.milkdrinkers:threadutil-bukkit:VERSION")
}

Shade Dependency

build.gradle.kts
plugins {
    id("com.gradleup.shadow") version "8.3.6"
}
 
tasks {
    shadowJar {
        relocate("io.github.milkdrinkers.threadutil", "yourpackage.threadutil")
    }
}

Gradle Groovy

Add Repository

build.gradle
repositories {
    mavenCentral()
}

Add Dependency

build.gradle
dependencies {
    implementation 'io.github.milkdrinkers:threadutil-bukkit:VERSION'
}

Shade Dependency

build.gradle
plugins {
    id 'com.gradleup.shadow' version '8.3.6'
}
 
tasks {
    shadowJar {
        relocate 'io.github.milkdrinkers.threadutil', 'yourpackage.threadutil'
    }
}

Maven

Add Dependency

pom.xml
<dependency>
    <groupId>io.github.milkdrinkers</groupId>
    <artifactId>threadutil-bukkit</artifactId>
    <version>VERSION</version>
</dependency>

Shade Dependency

pom.xml
<plugin>
    <groupId>org.apache.maven.plugins</groupId>
    <artifactId>maven-shade-plugin</artifactId>
    <version>3.5.0</version>
    <executions>
        <execution>
            <phase>package</phase>
            <goals>
                <goal>shade</goal>
            </goals>
        </execution>
    </executions>
    <configuration>
        <createDependencyReducedPom>false</createDependencyReducedPom>
        <relocations>
            <relocation>
                <pattern>io.github.milkdrinkers.threadutil</pattern>
                <shadedPattern>yourpackage.threadutil</shadedPattern>
            </relocation>
        </relocations>
    </configuration>
</plugin>
